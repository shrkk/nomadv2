```swift
import SwiftUI
import Photos

struct PhotoGalleryStrip: View {
    let startDate: Date
    let endDate: Date
    let boundingBox: (minLat: Double, maxLat: Double, minLon: Double, maxLon: Double)?

    @State private var thumbnails: [(id: String, image: UIImage)] = []
    @State private var isLoading = true
    @State private var permissionDenied = false

    var body: some View {
        Group {
            if permissionDenied {
                // Permission denied state per UI-SPEC Copywriting
                VStack(spacing: 8) {
                    Text("Allow photo access in Settings to see trip photos.")
                        .font(AppFont.caption())
                        .foregroundStyle(Color.Nomad.globeBackground.opacity(0.6))
                        .multilineTextAlignment(.center)
                    Button("Open Settings") {
                        if let url = URL(string: UIApplication.openSettingsURLString) {
                            UIApplication.shared.open(url)
                        }
                    }
                    .font(AppFont.caption())
                    .foregroundStyle(Color.Nomad.amber)
                }
                .padding(.horizontal, 16)
            } else if isLoading {
                ProgressView()
                    .frame(height: 80)
            } else if thumbnails.isEmpty {
                // Empty state per UI-SPEC Copywriting
                Text("No photos for this trip.")
                    .font(AppFont.caption())
                    .foregroundStyle(Color.Nomad.globeBackground.opacity(0.6))
                    .frame(maxWidth: .infinity)
                    .padding(.vertical, 16)
            } else {
                ScrollView(.horizontal, showsIndicators: false) {
                    LazyHStack(spacing: 8) {
                        ForEach(thumbnails, id: \.id) { item in
                            Image(uiImage: item.image)
                                .resizable()
                                .aspectRatio(contentMode: .fill)
                                .frame(width: 80, height: 80)
                                .clipShape(RoundedRectangle(cornerRadius: 8))
                            // D-18: Tap disabled in this phase — no gesture recognizer
                        }
                    }
                    .padding(.horizontal, 16)
                }
                .frame(height: 80)
            }
        }
        .task {
            await loadPhotos()
        }
    }

    private func loadPhotos() async {
        // Check authorization
        let status = PHPhotoLibrary.authorizationStatus(for: .readWrite)
        guard status == .authorized || status == .limited else {
            if status == .notDetermined {
                let newStatus = await PHPhotoLibrary.requestAuthorization(for: .readWrite)
                guard newStatus == .authorized || newStatus == .limited else {
                    await MainActor.run { permissionDenied = true; isLoading = false }
                    return
                }
            } else {
                await MainActor.run { permissionDenied = true; isLoading = false }
                return
            }
        }

        // Fetch PHAssets — two-pass approach per DETAIL-03 and DETAIL-04
        let options = PHFetchOptions()
        options.predicate = NSPredicate(
            format: "creationDate >= %@ AND creationDate <= %@",
            startDate as CVarArg, endDate as CVarArg
        )
        options.sortDescriptors = [NSSortDescriptor(key: "creationDate", ascending: true)]

        let allAssets = PHAsset.fetchAssets(with: .image, options: options)

        var matchedAssets: [PHAsset] = []
        allAssets.enumerateObjects { asset, _, _ in
            if let loc = asset.location, let bbox = boundingBox {
                // Pass 1: GPS bounding box filter
                let lat = loc.coordinate.latitude
                let lon = loc.coordinate.longitude
                if lat >= bbox.minLat && lat <= bbox.maxLat &&
                   lon >= bbox.minLon && lon <= bbox.maxLon {
                    matchedAssets.append(asset)
                }
            } else {
                // DETAIL-04: nil location — include via date-range-only fallback
                matchedAssets.append(asset)
            }
        }

        // Load thumbnails on background thread
        let targetSize = CGSize(width: 160, height: 160) // 2x for retina
        let imageOptions = PHImageRequestOptions()
        imageOptions.deliveryMode = .fastFormat
        imageOptions.isSynchronous = false
        imageOptions.isNetworkAccessAllowed = true

        var loaded: [(id: String, image: UIImage)] = []
        let manager = PHImageManager.default()

        for asset in matchedAssets.prefix(50) { // Cap at 50 thumbnails for performance
            let image = await withCheckedContinuation { (continuation: CheckedContinuation<UIImage?, Never>) in
                manager.requestImage(
                    for: asset,
                    targetSize: targetSize,
                    contentMode: .aspectFill,
                    options: imageOptions
                ) { image, info in
                    // Only use the final image (not the degraded placeholder)
                    let isDegraded = (info?[PHImageResultIsDegradedKey] as? Bool) ?? false
                    if !isDegraded {
                        continuation.resume(returning: image)
                    }
                }
            }
            if let img = image {
                loaded.append((id: asset.localIdentifier, image: img))
            }
        }

        await MainActor.run {
            thumbnails = loaded
            isLoading = false
        }
    }
}
```

Add a helper to compute the bounding box from route coordinates:

```swift
private func computeBoundingBox(from coords: [CLLocationCoordinate2D])
    -> (minLat: Double, maxLat: Double, minLon: Double, maxLon: Double)? {
    guard !coords.isEmpty else { return nil }
    let lats = coords.map(\.latitude)
    let lons = coords.map(\.longitude)
    // Add small padding (0.01 degrees ~ 1km) to catch nearby photos
    let padding = 0.01
    return (
        minLat: (lats.min() ?? 0) - padding,
        maxLat: (lats.max() ?? 0) + padding,
        minLon: (lons.min() ?? 0) - padding,
        maxLon: (lons.max() ?? 0) + padding
    )
}
```

The bounding box is computed from `routeCoordinates` (fetched from Firestore in Task 1's `.task`). Pass it to `PhotoGalleryStrip` after route loading completes. If route is still loading or empty, pass `nil` for boundingBox — the gallery will use date-range-only matching.

IMPORTANT per Research Pitfall note: PHImageManager.requestImage with `isSynchronous = false` uses a callback. The `withCheckedContinuation` pattern wraps this correctly. Filter out degraded (placeholder) images by checking `PHImageResultIsDegradedKey`.

NOTE: The `withCheckedContinuation` approach above may call the continuation twice since PHImageManager can call the result handler multiple times (once for degraded, once for final). To handle this safely, use a flag or only resume on non-degraded images. An alternative safer approach:

```swift
// Use a class-based image loader to handle the multi-callback pattern
for asset in matchedAssets.prefix(50) {
    await withCheckedContinuation { (cont: CheckedContinuation<Void, Never>) in
        var resumed = false
        manager.requestImage(for: asset, targetSize: targetSize,
                             contentMode: .aspectFill, options: imageOptions) { image, info in
            let degraded = (info?[PHImageResultIsDegradedKey] as? Bool) ?? false
            if !degraded && !resumed {
                resumed = true
                if let img = image {
                    loaded.append((id: asset.localIdentifier, image: img))
                }
                cont.resume()
            }
        }
    }
}
```

This ensures the continuation is only resumed once.
